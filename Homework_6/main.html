df_list = [df_sg, df_pr, df_bd, df_ps, df_dp]
name_list = [' ','','','',' ']
chart_list = []


for i in range(len(df_list)):
    df_i = df_list[i]
    
    lines = alt.Chart(df_i).mark_line().encode(
        x = alt.X('max():Q'),
        x2 = alt.X2('min():Q'),
        y = alt.Y(':N', 
                  sort=alt.Sort(field='', op='max', order='descending'))
        )

    points = alt.Chart(df_i).mark_circle(size = 100).encode(
        x = alt.X(':Q', title = ", %"),
        y = alt.Y(':N',  title = '', 
                  sort=alt.Sort(field='', op='max', order='descending')),
        color  = alt.Color(':N', title = ""),
        tooltip = [ alt.Tooltip(':N', title = ''),
                    alt.Tooltip(':N', title = '' ),
                    alt.Tooltip(':Q', title = '' ) ]
    )

    chart = alt.layer(lines, points
                     ).properties(width = 400, height = 500,
                                  title = { "text": '   2017    2010 ', 
                                            "subtitle": name_list[i],
                                            "subtitleFontSize": 14,
                                            "subtitleFontWeight": 'bold'}          
                     )  
    
    chart_list.append(chart)
